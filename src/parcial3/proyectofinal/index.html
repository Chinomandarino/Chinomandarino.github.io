Consulta 7

Mostrar la asignatura que tiene más alumnos inscritos en total

SELECT asignatura.nombre,
COUNT(matricula.idAlumno) AS total_inscritos
FROM asignatura
JOIN matricula ON asignatura.idAsignatura =
matricula.idAsignatura
GROUP BY asignatura.nombre
ORDER BY total_inscritos DESC;

Consulta 9

Obtener listado de cursos, con asignatura, y el maestro que las imparte

SELECT
curso.idCurso,
curso.nombreDescriptivo AS nombreCurso,
asignatura.idAsignatura,
asignatura.nombre AS nombreAsignatura,
profesor.idProfesor,
profesor.nombre AS nombreProfesor,
profesor.apellido1,
profesor.apellido2
FROM curso
JOIN asignatura ON curso.idCurso =
asignatura.curso
JOIN impartir ON asignatura.idAsignatura =
impartir.idAsignatura
JOIN profesor ON impartir.idProfesor =
profesor.idProfesor;

Consulta 10

Obtener listado de alumnos, con asignaturas que cursa, profesor que las imparte y al curso que pertenece

SELECT
alumno.idAlumno,
alumno.nombre AS nombreAlumno,
alumno.apellido1,
alumno.apellido2,
asignatura.idAsignatura,
asignatura.nombre AS nombreAsignatura,
profesor.idProfesor,
profesor.nombre AS nombreProfesor,
profesor.apellido1 AS apellidoProfesor1,
profesor.apellido2 AS apellidoProfesor2,
curso.idCurso,
curso.nombreDescriptivo AS nombreCurso
FROM alumno
JOIN matricula ON alumno.idAlumno =
matricula.idAlumno
JOIN asignatura ON matricula.idAsignatura =
asignatura.idAsignatura
JOIN impartir ON asignatura.idAsignatura =
impartir.idAsignatura
JOIN profesor ON impartir.idProfesor =
profesor.idProfesor
JOIN curso ON asignatura.curso = curso.idCurso;
