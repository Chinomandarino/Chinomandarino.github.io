Mostrar la asignatura que tiene mas alumnos inscritos en total

SELECT asignatura.nombre,
COUNT(matricula.idAlumno) AS total_inscritos
FROM asignatura
JOIN matricula ON asignatura.idAsignatura =
matricula.idAsignatura
GROUP BY asignatura.nombre
ORDER BY total_inscritos DESC;
